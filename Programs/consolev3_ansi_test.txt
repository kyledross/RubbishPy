' Send the letters A through Z to the console with no CR/LF
' Send enough to cause the cursor to wrap to the next line

jmp main

' color escape routines ----------------------'

escape: lr 1 27
mrm 1 1024
' [
lr 1 91
mrm 1 1024
rtn

' 31 red
red: lr 1 51
mrm 1 1024
lr 1 49
mrm 1 1024
rtn

' 32 green
green: lr 1 51
mrm 1 1024
lr 1 50
mrm 1 1024
rtn

' 34 blue
blue: lr 1 51
mrm 1 1024
lr 1 52
mrm 1 1024
rtn

' 37 white
white: lr 1 51
mrm 1 1024
lr 1 55
mrm 1 1024
rtn

' m end of escape sequence
m: lr 1 109
mrm 1 1024
rtn

' --------------------------------------------------'
' display text

' char x
main: nop
' call escape
' call white
' call m
' lr 1 120
' mrm 1 1024

' char r
call escape
call red
call m
lr 1 114
mrm 1 1024

' char g
call escape
call green
call m
lr 1 103
mrm 1 1024

' char b
call escape
call blue
call m
lr 1 98
mrm 1 1024

' halt
jmp main
